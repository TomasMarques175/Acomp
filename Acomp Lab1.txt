.data

	u1: .word 1		#f(1)
	u2: .word 1		#f(2)

.text
	la x10,u1		#Load do endereço de f(1)

	li x14, 2		#inicializa a 2 pois já temos os 2 primeiros valores da sucessão
	li x15, 10		#pretende-se fazer a contagem até ao 10 termo da sucessão

for:
	bge x14,x15,end	#repete o ciclo até atingir o 10ºtermo da sucessão
	lw x11,0(x10)	#Load do valor da memoria referente a f(n-1)
	lw x12,4(x10)	#Load do valor da memoria referente a f(n)
	add x13,x12,a1	#Soma de f(n-1) com f(n)
	sw x13,8(x10)	#Store do valor da soma no endereço X13

	addi x14,x14,1	#contagem do numero de ciclos
	addi x10,x10,4	#salta para a posição de memoria seguinte

	j for

end:
	li x17,10		#termina o programa
	ecall
